          {{'\\STATE $f=\\mathcal{f}_0$'}}
          {{'\\STATE \\textbf{Se determină rețeaua reziduală} $G(f)=(N,A(f))$'}}

          {{'\\ENDWHILE'}}
        {{'\\end{algorithmic}'}}
